6   package kayttoliittyma;
7   
8   import java.awt.Container;
9   import java.awt.GridLayout;
10  import javax.swing.ImageIcon;
11  import javax.swing.JButton;
12  import javax.swing.JFrame;
13  import javax.swing.WindowConstants;
14  import kayttoliittyma.kuuntelijat.KorotusKuuntelija;
15  import static logiikka.peli.Maa.MUSTA;
16  import logiikka.peli.Pelaaja;
17  
18  /**
19   * Näyttää valikon, josta valitaan miksi nappulaksi korotetaan vastustajan
20   * perusriville päässyt sotilas
21   *
22   * @author elias
23   */
24  public class KorotusValikko implements Runnable {
25  
26      private Kayttoliittyma kayttoliittyma;
27      private JFrame ruutu;
28      private Pelaaja korotettavaPelaaja;
29  
30      public Pelaaja getKorotettavaPelaaja() {
31          return korotettavaPelaaja;
32      }
33  
34      public Kayttoliittyma getKayttoliittyma() {
35          return kayttoliittyma;
36      }
37  
38      public KorotusValikko(Kayttoliittyma kayttoliittyma, Pelaaja korotettava) {
39          this.kayttoliittyma = kayttoliittyma;
40          this.korotettavaPelaaja = korotettava;
41      }
42  
43      @Override
44      public void run() {
45          ruutu = new JFrame("Miksi korotat?");
46          ruutu.setBounds(this.kayttoliittyma.getFrame().getBounds());
47  
48          ruutu.setAlwaysOnTop(true);
49          ruutu.setDefaultCloseOperation(WindowConstants.DO_NOTHING_ON_CLOSE);
50  
51          luoKomponentit(ruutu.getContentPane());
52  
53          ruutu.pack();
54          ruutu.setVisible(true);
55      }
56  
57      /**
58       * luo näytettävät valintanapit ja lisää niihin kuuntelijat
59       *
60       * @param container
61       */
62      private void luoKomponentit(Container container) {
63  
64          GridLayout layout = new GridLayout(1, 4);
65          container.setLayout(layout);
66  
67          String polku;
68          if (this.korotettavaPelaaja.getMaa() == MUSTA) {
69              polku = "nappulakuvat/musta/";
70          } else {
71              polku = "nappulakuvat/valkoinen/";
72  
73          }
74  
75          JButton kuningatarNappi = new JButton(new ImageIcon(polku + "kuningatar.png"));
76          kuningatarNappi.addActionListener(new KorotusKuuntelija(this, 'q'));
77          container.add(kuningatarNappi);
78  
79          JButton torniNappi = new JButton(new ImageIcon(polku + "torni.png"));
80          torniNappi.addActionListener(new KorotusKuuntelija(this, 't'));
81          container.add(torniNappi);
82  
83          JButton lahettiNappi = new JButton(new ImageIcon(polku + "lahetti.png"));
84          lahettiNappi.addActionListener(new KorotusKuuntelija(this, 'l'));
85          container.add(lahettiNappi);
86  
87          JButton ratsuNappi = new JButton(new ImageIcon(polku + "ratsu.png"));
88          ratsuNappi.addActionListener(new KorotusKuuntelija(this, 'r'));
89          container.add(ratsuNappi);
90  
91      }
92  
93      public JFrame getRuutu() {
94          return ruutu;
95      }
96  
97  }
