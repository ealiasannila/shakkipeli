6   package kayttoliittyma.kuuntelijat;
7   
8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import kayttoliittyma.KorotusValikko;
11  import logiikka.peli.Pelaaja;
12  
13  /**
14   * Kuuntelee korotusvalikon nappeja
15   *
16   * @author elias
17   */
18  public class KorotusKuuntelija extends KayttoliittymanTuntevaLuokka implements ActionListener {
19  
20      private Pelaaja korotettava;
21      private KorotusValikko valikko;
22      private char miksiKorotetaan;
23  
24      public KorotusKuuntelija(KorotusValikko valikko, char miksiKorotetaan) {
25          super(valikko.getKayttoliittyma());
26  
27          this.valikko = valikko;
28          this.korotettava = valikko.getKorotettavaPelaaja();
29          this.valikko = valikko;
30          this.miksiKorotetaan = miksiKorotetaan;
31      }
32  
33      /**
34       * pyytää peliä korottamaan korotusta odottavan sotilaan valituksi
35       * upseeriksi, sulkee korotusikkunan
36       *
37       * @param ae
38       */
39      @Override
40      public void actionPerformed(ActionEvent ae) {
41          this.kayttoliittyma.getPeliHallinta().getPeli().korotaSotilas(this.korotettava, miksiKorotetaan);
42  
43          this.kayttoliittyma.getPeliPiirto().setSotilaanKorotusKesken(false);
44          this.kayttoliittyma.getPeliPiirto().repaint();
45  
46          this.valikko.getRuutu().dispose();
47      }
48  
49  }
